class Tree:
     name: str
     height: float
     leafs: int

     def __init__(self, name: str, height: float, leafs: int) -> None:
         self.name = name
         self.height = height
         self.leafs = leafs

     def grow_up(self, height: float) -> float:
         return height

     def grow_wide(self, leafs: int) -> int:
         return leafs

     def show(self, height: float, leafs: int, name: str):
         return f'wysokoœæ drzewa: {self.height}, liczba lisci: {self.leafs}'
